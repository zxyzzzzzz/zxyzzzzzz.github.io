#include “malloc.h”

#include “stdio.h”

#define ADD 10

typedef int status;

typedef char SelemType; /上两条语句将int 和char 定义了一个新名称/

typedef struct{

SelemType *base;


SelemType *top;


int stacksize;

} SqStack;

status InitStack(SqStack &S){

S.base=(SelemType *)malloc(100*sizeof(SelemType));//构建一个100层的空栈 


S.top=S.base;       //令S的顶端指针指向底部


S.stacksize=100;


return 1;

} status Push(SqStack &S,SelemType e){

//入栈操作


if(S.top-S.base>=S.stacksize){/*判断栈满则追加空间*/


    S.base=(SelemType *)realloc(S.base,(S.stacksize

+ADD)*sizeof(SelemType));//用动态分布的方式增加10层

    S.top=S.base+S.stacksize; //存疑！！！！！！！！！


    S.stacksize+=ADD ;


} 


*S.top++ =e;


return 1;

}

status pop(SqStack &S,SelemType &e){

//出栈操作 删除S的栈顶元素用e返回其值


if(S.top==S.base) return 0;


e=*--S.top;


return 1;

}
